% Name formatting
\newcommand{\namesection}[3]{
  \centering{
    \fontsize{30pt}{40pt}\selectfont
    \textbf{#1 #2}
  }\\[4pt]
  \centering{
    \color{headings}
    \fontsize{10pt}{14pt}\selectfont
    #3
  }\\[4pt]
  \noindent{\color{headings}\rule{\textwidth}{0.4pt}}
  \vspace{-10pt}
}

% Section formatting
\titleformat{\section}{
  \color{headings}\scshape\fontsize{14pt}{18pt}\selectfont\raggedright\uppercase
}{}{0em}{}

% Subsection formatting
\titleformat{\subsection}{
  \color{subheadings}\fontsize{11pt}{13pt}\selectfont\bfseries\uppercase
}{}{0em}{}

\titlespacing{\section}{0pt}{6pt}{4pt}
\titlespacing{\subsection}{0pt}{4pt}{2pt}

% Run subsection (inline title for experience)
\newcommand{\runsubsection}[1]{
  \color{subheadings}\fontsize{11pt}{13pt}\selectfont\bfseries\uppercase{#1}\normalfont
}

% Descriptor (pipe + role name)
\newcommand{\descript}[1]{
  \color{subheadings}\raggedright\scshape\fontsize{10pt}{12pt}\selectfont{#1\\}\normalfont
}

% Location formatting
\newcommand{\location}[1]{
  \color{headings}\raggedright\fontsize{9pt}{11pt}\selectfont{#1\\}\normalfont
}

% Tight item list
\newenvironment{tightitemize}{
  \vspace{2pt}\begin{itemize}[leftmargin=12pt]\itemsep1pt \parskip0pt \parsep0pt
}{
  \end{itemize}\vspace{-\topsep}
}

\newcommand{\sectionsep}{\vspace{6pt}}
